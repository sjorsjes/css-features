@container()

#

Container queries are an alternative to media queries, which apply styles to elements based on viewport size or other device characteristics.

@scope()

#

Allows CSS rules to be scoped to part of the document, with upper and lower limits described by selectors. 

  
  <div class="motorbike">
    <div class="bicycle">
      <div class="tricycle">
        <div class="item"></div>
      </div>
    </div>
  </div>
  


@scope (.tricycle) {
  .item {
    background: url("tricycle.jpg");
  }
}

@scope (.bicycle) {
  .item {
    background: url("bicycle.jpg");
  }
}

@scope (.motorbike) {
  .item {
    background: url("motorbike.jpg");
  }
}
